START

IMPORT:
  - fmt
  - smtp
  - os

DEFINE STRUCT EmailService:
  smtpHost
  smtpPort
  senderEmail
  senderPassword


FUNCTION NewEmailService:

  AMBIL konfigurasi dari environment:
    smtpHost
    smtpPort
    senderEmail
    senderPassword

  RETURN instance EmailService


FUNCTION SendResetCode(recipientEmail, resetCode):

  SET subject = "Kode Reset Password Pil-Time"

  BUAT isi email (HTML):
    - header
    - pesan reset password
    - tampilkan resetCode
    - info masa berlaku 15 menit

  SET authentication SMTP menggunakan:
    senderEmail dan senderPassword

  SET alamat server SMTP:
    smtpHost + smtpPort

  KIRIM email ke recipientEmail

  JIKA gagal:
    RETURN error

  RETURN sukses

END
