START

IMPORT:
  - errors
  - os
  - time
  - jwt

DEFINE STRUCT Claims:
  id
  email
  role
  registeredClaims

SET defaultSecret = "your-secret-key-change-this-in-production"

FUNCTION getSecret:

  AMBIL JWT_SECRET dari environment

  JIKA kosong:
    RETURN defaultSecret

  RETURN secret

FUNCTION GenerateNakesToken(nakesID, email):

  BUAT claims:
    id = nakesID
    email
    role = "admin"
    expired = sekarang + 24 jam
    issuedAt = sekarang

  BUAT token dengan claims

  SIGN token dengan secret

  RETURN token

FUNCTION GeneratePasienToken(pasienID, email):

  BUAT claims:
    id = pasienID
    email
    role = "pasien"
    expired = sekarang + 24 jam
    issuedAt = sekarang

  BUAT token dengan claims

  SIGN token dengan secret

  RETURN token

FUNCTION ValidateToken(tokenString):

  PARSE token dengan claims

  JIKA error:
    RETURN error

  JIKA token tidak valid:
    RETURN error "invalid token"

  RETURN claims

END
