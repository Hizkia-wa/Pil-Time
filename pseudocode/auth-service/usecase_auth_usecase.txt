DEFINE STRUCT AuthUsecase:
  nakesRepo
  pasienRepo
  emailService

FUNCTION NewAuthUsecase(nakesRepo, pasienRepo):

  BUAT instance AuthUsecase:
    emailService = EmailService baru

  RETURN AuthUsecase

FUNCTION LoginNakes(req):

  AMBIL data nakes berdasarkan email

  JIKA tidak ditemukan:
    RETURN error "email atau password salah"

  VERIFIKASI password

  JIKA salah:
    RETURN error "email atau password salah"

  GENERATE token JWT

  JIKA gagal:
    RETURN error "gagal membuat token"

  RETURN response:
    token
    data nakes
    message "Login berhasil"

FUNCTION RegisterPasien(req):

  CEK email sudah ada:
    JIKA ada:
      RETURN error "email sudah terdaftar"

  CEK NIK sudah ada:
    JIKA ada:
      RETURN error "NIK sudah terdaftar"

  HASH password

  JIKA gagal:
    RETURN error "gagal mengenkripsi password"

  PARSE tanggal lahir

  JIKA format salah:
    RETURN error "format tanggal salah"

  BUAT objek pasien

  SIMPAN ke database

  JIKA gagal:
    RETURN error "gagal menyimpan data"

  RETURN response registrasi berhasil

FUNCTION LoginPasien(req):

  AMBIL data pasien berdasarkan email

  JIKA tidak ditemukan:
    RETURN error "email atau password salah"

  VERIFIKASI password

  JIKA salah:
    RETURN error "email atau password salah"

  GENERATE token JWT

  JIKA gagal:
    RETURN error "gagal membuat token"

  RETURN response login berhasil

FUNCTION ForgotPassword(req):

  CEK email pasien

  JIKA tidak ditemukan:
    RETURN error "email tidak terdaftar"

  GENERATE kode acak 6 digit

  SET waktu kadaluarsa = sekarang + 15 menit

  SIMPAN kode dan expiry ke database

  JIKA gagal:
    RETURN error "gagal menyimpan kode"

  KIRIM kode ke email

  JIKA gagal:
    RETURN error "gagal mengirim email"

  RETURN message sukses

FUNCTION VerifyResetCode(req):

  CEK email pasien

  JIKA tidak ditemukan:
    RETURN error

  JIKA tidak ada kode reset:
    RETURN error

  JIKA kode kadaluarsa:
    RETURN error

  JIKA kode tidak cocok:
    RETURN error

  RETURN message kode valid

FUNCTION ResetPassword(req):

  VERIFIKASI kode reset terlebih dahulu

  JIKA gagal:
    RETURN error

  HASH password baru

  JIKA gagal:
    RETURN error

  UPDATE password di database

  JIKA gagal:
    RETURN error

  RETURN message password berhasil diubah

FUNCTION ValidateToken(token):

  VALIDASI token menggunakan utils

  JIKA tidak valid:
    RETURN valid = false

  RETURN:
    valid = true
    userID
    email
    role

FUNCTION generateRandomCode:

  BUAT angka acak 6 digit

  RETURN kode
