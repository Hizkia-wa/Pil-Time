START

IMPORT:
  - domain
  - outbound interface
  - time
  - gorm

DEFINE STRUCT pasienRepo:
  db


FUNCTION NewPasienRepo(db):

  BUAT instance pasienRepo dengan db

  RETURN sebagai PasienRepository


FUNCTION Create(pasien):

  SIMPAN data pasien ke database

  RETURN hasil error (jika ada)


FUNCTION GetByEmail(email):

  DECLARE data pasien

  QUERY database:
    cari pasien berdasarkan email

  JIKA error:
    RETURN null dan error

  RETURN data pasien


FUNCTION GetByNIK(nik):

  DECLARE data pasien

  QUERY database:
    cari pasien berdasarkan nik

  JIKA error:
    RETURN null dan error

  RETURN data pasien


FUNCTION GetByID(id):

  DECLARE data pasien

  QUERY database:
    cari pasien berdasarkan id

  JIKA error:
    RETURN null dan error

  RETURN data pasien


FUNCTION UpdateResetCode(email, code, expiryTime):

  UPDATE data pasien berdasarkan email:
    set reset_code = code
    set reset_code_expiry = expiryTime

  RETURN hasil error


FUNCTION UpdatePassword(email, hashedPassword):

  UPDATE data pasien berdasarkan email:
    set password = hashedPassword
    set reset_code = null
    set reset_code_expiry = null

  RETURN hasil error

END
