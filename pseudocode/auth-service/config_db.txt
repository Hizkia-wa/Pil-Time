START

IMPORT:
  - fmt
  - log
  - os
  - godotenv
  - postgres driver
  - gorm

FUNCTION InitPostgres:

  1. LOAD FILE ENV:
     error = load .env

     JIKA error:
       TAMPILKAN warning
       "env file tidak ditemukan, gunakan environment variables"

  2. BUAT DATABASE CONNECTION STRING (DSN):
     dsn = gabungkan:
       host = DB_HOST
       user = DB_USER
       password = DB_PASSWORD
       dbname = DB_NAME
       port = DB_PORT
       sslmode = DB_SSLMODE

  3. KONEKSI KE DATABASE:
     db, error = buka koneksi postgres menggunakan dsn

     JIKA error:
       HENTIKAN program (panic)
       tampilkan pesan "Gagal koneksi database"

  4. JIKA BERHASIL:
     tampilkan log "Database connected"

  5. RETURN db

END
