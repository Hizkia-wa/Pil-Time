START

IMPORT MODULE:
  config, httpHandler, persistence, usecase, gin, log

MAIN PROCESS:

  [1] INITIALIZE DATABASE
      db ← InitPostgres()

  [2] SETUP DEPENDENCIES

      REPOSITORY LAYER:
        nakesRepo ← NewNakesRepo(db)
        pasienRepo ← NewPasienRepo(db)

      USECASE LAYER:
        authUsecase ← NewAuthUsecase(nakesRepo, pasienRepo)

      HANDLER LAYER:
        authHandler ← NewAuthHandler(authUsecase)

  [3] CONFIGURE ROUTER
      router ← gin.New()

      APPLY MIDDLEWARE:
        - Logger
        - Recovery
        - CORS Middleware

  [4] DEFINE ENDPOINT: HEALTH CHECK
      GET "/health":
        RETURN JSON:
          status ← "Auth Service running"
          port ← 8081

  [5] DEFINE ROUTES: AUTHENTICATION

      BASE ROUTE "/auth":

        → SUBROUTE "/nakes":
            POST "/login" → Handler: LoginNakes

        → SUBROUTE "/pasien":
            POST "/register" → Handler: RegisterPasien
            POST "/login" → Handler: LoginPasien
            POST "/forgot-password" → Handler: ForgotPassword
            POST "/verify-reset-code" → Handler: VerifyResetCode
            POST "/reset-password" → Handler: ResetPassword

        → GET "/validate" → Handler: ValidateToken

  [6] START SERVER
      PRINT "Auth Service starting on port 8081..."

      RUN server at port 8081

      IF server fails:
        PRINT error
        TERMINATE program


CORS MIDDLEWARE PROCESS:

  SET HEADER:
    Allow-Origin ← request origin
    Allow-Methods ← GET, POST, PUT, DELETE, OPTIONS
    Allow-Headers ← Origin, Content-Type, Authorization, Accept
    Allow-Credentials ← true
    Max-Age ← 86400

  IF request method = OPTIONS:
    RETURN status 204 (Preflight response)
    STOP process

  CONTINUE to next handler

END
